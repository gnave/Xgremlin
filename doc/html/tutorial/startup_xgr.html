An example .xgremlinrc file

# local XGremlin initialization file
alias exit end               # Set an alias for a command or 
                             # set of commands 
set noecho                   # Stops echoing the commands
button 1 edit "plot all" "plot all"   # Re-label a button
button 2 plot "add line" C+a  # Re-label another button
break                        #  exits the file and returns control
                             #  to the xgremlin prompt
